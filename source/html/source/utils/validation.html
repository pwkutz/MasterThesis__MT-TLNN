class Validator:
    def __init__(self, train_test_dataset, troll_raw_data_location = "H:/NOT_SAVED_TO_BACKUP/TROLL_data/vlad/", modelica_data_from_troll_location = "D:/Vlad/data/all_runs_from_troll_unvalidated.h5", json_w_start_times = 'parameters/analyazable_starttime.json', json_w_stop_times = "parameters/runs_stop_time.json"):
        self.train_test_dataset = train_test_dataset

    def validate_dataset(self, method, train_or_test):
        print("Input variables")
        pd.DataFrame(self.train_test_dataset.dataset[method].train_test_dataset[train_or_test]["X"]).describe()
        print("Output variables")
        pd.DataFrame(self.train_test_dataset.dataset[method].train_test_dataset[train_or_test]["y"]).describe()

    def plot_against_time(self, feature_of_interest, run_ids_list):
        num_of_ids = min(9, len(run_ids_list)) # don't plot more than 9 runs
        num_columns = int(np.sqrt(num_of_ids))
        fig, ax = plt.subplots(nrows=num_of_ids//3, ncols=3, figsize=(20, (num_of_ids//3)*7))#, sharey = True, sharex = True)
        random_choice = np.random.choice(run_ids_list, num_of_ids, replace=False) 
        for i in range(num_of_ids):
            run_id = random_choice[i]
            if num_of_ids > 3:
                ax_ = ax[i//3, i%3]
            else:
                ax_ = ax[i%3]
            current_X_terra, current_y_terra, current_strart_time_terra, current_stop_time_terra = extract_data_from_run("terra", run_id, self.troll_raw_data_location, self.modelica_data_from_troll_location, self.json_w_start_times, self.json_w_stop_times)
            current_X_scm, current_y_scm, current_strart_time_scm, current_stop_time_scm = extract_data_from_run("scm", run_id, self.troll_raw_data_location, self.modelica_data_from_troll_location, self.json_w_start_times, self.json_w_stop_times)
            current_X_troll, current_y_troll, current_strart_time_troll, current_stop_time_troll = extract_data_from_run("troll", run_id, self.troll_raw_data_location, self.modelica_data_from_troll_location, self.json_w_start_times, self.json_w_stop_times)
            x_terra = np.linspace(current_strart_time_terra, current_stop_time_terra, current_X_terra.shape[0])
            x_scm = np.linspace(current_strart_time_scm, current_stop_time_scm, current_X_scm.shape[0])
            x_troll = np.linspace(current_strart_time_troll, current_stop_time_troll, current_X_troll.shape[0])
            feature_of_interest
            ax_.plot(x_terra, feature_of_interest, label = "TerRA",c = "red")
            ax_.plot(x_scm, feature_of_interest, label = "SCM",  c = "blue")
            ax_.plot(x_troll, feature_of_interest, label = "TROLL",  c = "green")
            ax_.set_xlabel("time, [s]")
            ax_.set_ylabel(feature_of_interest + " , [m]")
            ax_.legend()
            ax_.set_title("TROLL run with ID: " + str(run_id))
        plt.show()

        self.plot_against_time()
        self.plot_against_time()
        self.plot_against_time()

class SynthValidator:
    def __init__(self, num_of_exp, modelica_data_from_troll_location = "D:/Vlad/data/all_runs_synth.h5"):
        data = h5py.File(modelica_data_from_troll_location, 'r')
        self.run_length = 201
        scm_X = []
        scm_y = []
        terra_X = []
        terra_y = []
        for i in range(num_of_exp):
            scm_X.append(np.array([data[str(i)]["scm"]["input"][x][1][:12] for x in range(int(len(data[str(0)]["scm"]["input"])))]))
            scm_y.append(np.array([data[str(i)]["scm"]["output"][x][1][:6] for x in range(int(len(data[str(0)]["scm"]['output'])))]))
        for i in range(num_of_exp):
            terra_X.append(np.array([data[str(i)]["terra"]["input"][x][1][:12] for x in range(int(len(data[str(0)]["scm"]['input'])))]))
            terra_y.append(np.array([data[str(i)]["terra"]["output"][x][1][:6] for x in range(int(len(data[str(0)]["scm"]['output'])))]))

        self.scm_X = np.concatenate(scm_X, axis=0)
        self.scm_y = np.concatenate(scm_y, axis=0)
        self.terra_X = np.concatenate(terra_X, axis=0)
        self.terra_y = np.concatenate(terra_y, axis=0)
        
        self.dataset = {"terra":{"X": self.terra_X, "y": self.terra_y},
                        "scm":{"X": self.scm_X, "y": self.scm_y}}

        # variables that will be used for plotting
        self.force_x_scm = self.scm_y[:, 0]
        self.force_z_scm = self.scm_y[:, 2]
        self.force_x_terra = self.terra_y[:, 0]
        self.force_z_terra = self.terra_y[:, 2]

    def sinkage_against_time(self):
        plt.figure(figsize=(12,7))
        for i in range(10):
            plt.plot(range(self.run_length), self.scm_X[(i*self.run_length):((i+1)*self.run_length), 2], c = 'blue', alpha=0.2)
            plt.plot(range(self.run_length), self.terra_X[(i*self.run_length):((i+1)*self.run_length), 2], c = 'red', alpha=0.2)
        plt.xlabel("timestamp, [s]")
        plt.ylabel("sinkage, [m]")
        plt.legend(["SCM", "TerRA"])

    def traction_against_time(self):
        plt.figure(figsize=(15,10))
        for i in range(10):
            plt.plot(range(self.run_length), self.force_x_scm[(i*self.run_length):((i+1)*self.run_length)], c = 'blue', alpha=0.2)
            plt.plot(range(self.run_length), self.force_x_terra[(i*self.run_length):((i+1)*self.run_length)], c = 'red', alpha=0.2)
        plt.xlabel("timestamp, [s]")
        plt.ylabel("traction force, [N]")
        plt.legend(["SCM", "TerRA"])

    def normal_against_time(self):        
        plt.figure(figsize=(15,10))
        for i in range(10):
            plt.plot(range(self.run_length), self.force_x_scm[(i*self.run_length):((i+1)*self.run_length)], c = 'blue', alpha=0.2)
            plt.plot(range(self.run_length), self.force_z_terra[(i*self.run_length):((i+1)*self.run_length)], c = 'red', alpha=0.2)
        plt.xlabel("timestamp, [s]")
        plt.ylabel("normal force, [N]")
        plt.ylim(-5, 100)
        plt.legend(["SCM", "TerRA"])
